#Create embedding lookup to connect vectorizer with pre-trained embedding
#EMbedding matrix is the weights for embedding layer
def create_embedding_mat():
    vocab = text_vectorizer.get_vocabulary()[2:]
    #vocab = [word.decode('utf-8') for word in vocab]
    word_index = dict(zip(vocab, range(2, len(vocab))))
    embedding_matrix = np.zeros((vocab_size, 200))
    for word, i in word_index.items():
        embedding_vector = vectors.loc[word].values
        embedding_matrix[int(i)] = embedding_vector
    return embedding_matrix

embedding_matrix = create_embedding_mat()

# Build the model.

hidden_1 = 256
hidden_2 = 128

#Build encoder.
review_inputs = layers.Input(shape=(SEQU_LENGTH,200), dtype='float32', name='reviews')
x, state_h1, state_c1 = layers.LSTM(hidden_1, 
                activation='relu',
                return_state=True,
                return_sequences=True,
                name='Encoder_1')(review_inputs) # Return states to init decoder with it.
encoder_states1 = [state_h1, state_c1]
encoder_outputs, state_h2, state_c2 = layers.LSTM(hidden_2,
                 activation='relu',
                 return_state=True,
                 name='Encoder_2')(x)
encoder_states2 = [state_h2, state_c2]

# Build decoder.
x1 = layers.RepeatVector(SEQU_LENGTH)(encoder_outputs)
x1 = layers.LSTM(hidden_2,activation='relu',
                 return_sequences=True,
                 name='Decoder_1')(x1, initial_state=encoder_states2)
x1 = layers.LSTM(hidden_1,
                activation='relu',
                return_sequences=True,
                name='Decoder_2')(x1, initial_state=encoder_states1)
decoder_outputs = layers.TimeDistributed(layers.Dense(200, activation=None))(x1)

# Assemble into Autoencoder.
autoencoder = keras.Model(inputs=review_inputs, outputs=decoder_outputs)
autoencoder.summary()

def reset_lr(old_model, lr):
    weights = old_model.get_weights()
    new_model = keras.Model(review_inputs, decoder_outputs, name="autoencoder")
    new_model.compile(
        optimizer=tf.keras.optimizers.Adam(learning_rate=lr),  # Optimizer
        # Loss function to minimize
        loss='mse',
        # List of metrics to monitor
        metrics=['mse'])
    new_model.set_weights(weights)
    return new_model

##Func to turn word vec into the closest word.
def most_similar(v, N=10):
    #sims = vectors.values.dot(v)
    sims = cosine_similarity(vectors.values,v.reshape(1,-1))
    sims = heapq.nlargest(N, zip(sims,vectors.index.values))  
    return sims
