import re

def word_counter(df):
    word_lists = []
    for review in df.iloc[:,0]:
        word_lists.append(review_to_wordlist(review, remove_stopwords=False))

    word_count = []
    for review in word_lists:
        word_count.append(len(review))
    #Normalize counts
    counts = np.array(word_count)
    counts = (counts - counts.mean())/counts.std()
    return counts

#Create embedding lookup to connect vectorizer with pre-trained embedding
#EMbedding matrix is the weights for embedding layer
def create_embedding_mat():
    vocab = text_vectorizer.get_vocabulary()
    vocab = [word.decode('utf-8') for word in vocab]

    word_index = dict(zip(vocab, range(2, len(vocab))))

    embedding_matrix = np.zeros((vocab_size, 200))
    for word, i in word_index.items():
        embedding_vector = vectors.loc[word].values
        embedding_matrix[int(i)] = embedding_vector
    return embedding_matrix

def compile_model(model):

    lr_schedule = tf.keras.optimizers.schedules.ExponentialDecay(initial_learning_rate,
                                    decay_steps=200,

        optimizer=tf.keras.optimizers.Adam(learning_rate=initial_learning_rate),  # Optimizer
        # Loss function to minimize
        loss=tf.keras.losses.BinaryCrossentropy(from_logits=False),
        # List of metrics to monitor
        metrics=['accuracy','AUC','Recall','Precision'])

    return None

def fit_model(model, X, y, epochs, tag):

    #Init TensorBoard
    now = datetime.datetime.now().strftime("%m%d-%H%M%S")
    log_dir = "./neural_net/tf_logs/rnn_logs/" + tag + now
    tensorboard_callback = tf.keras.callbacks.TensorBoard(log_dir=log_dir,
                                                          histogram_freq=1,
                                                          update_freq='batch')
    #Init Checkpoints
    ckpt_dir = log_dir +'/weights'
    checkpoints = tf.keras.callbacks.ModelCheckpoint(
                    ckpt_dir, monitor='val_loss', save_best_only=True,
                    save_weights_only=False, mode='auto', save_freq='epoch')

    #Init hparam recordings
    hparam_cb = hp.KerasCallback(log_dir+'/validation/', hparams=hparams)

    neg, pos = pd.Series(y.reshape(-1,)).value_counts(sort=False).values
    total = neg+pos
    weight_for_0 = (1 / neg)*(total)/2.0
    weight_for_1 = (1 / pos)*(total)/2.0
    class_weight={0: weight_for_0, 1: weight_for_1}

    history = model.fit(
    X,
    y,
    batch_size=512,
    epochs=epochs,
    validation_data=({'reviews':X_val, 'length':val_counts},y_val),
    verbose=1,
    #callbacks=[tensorboard_callback,checkpoints,hparam_cb],

    return history

import matplotlib.pyplot as plt
from sklearn.metrics import roc_curve, auc, confusion_matrix
from sklearn.metrics import precision_recall_curve, f1_score, precision_score, recall_score

%matplotlib inline
def curves(model,X,y):

    y_probas = model.predict(X, batch_size=512)
    y_pred = (y_probas > 0.5).astype("int32")

    def roc(model, X, y):
        fpr, tpr, thres = roc_curve(y, y_probas)
        plt.plot(fpr,tpr)
        plt.plot([0,1],[0,1], linestyle='--')
        plt.title(f'AUC:{auc(fpr,tpr)}')
        print(f'AUC:{auc(fpr,tpr)}')


    def pr_curve(model, X, y):

        pr, re, thres = precision_recall_curve(y, y_probas)
        f1 = f1_score(y, y_pred)
        plt.plot(re,pr)
        plt.title(f'F1:{f1}')
        plt.xlabel(f'Recall: {recall_score(y, y_pred)}')
        plt.ylabel(f'Precision: {precision_score(y, y_pred)}')
        print(f'F1:{f1}')


    print(confusion_matrix(y, y_pred))

    f, ax = plt.subplots(1,2,figsize=(10,4), sharex=True, sharey=True)
    plt.subplot(1,2,1)
    roc(model,X,y)

    plt.subplot(1,2,2)
    pr_curve(model,X,y)
